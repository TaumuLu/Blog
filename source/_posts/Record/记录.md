轻量级服务端渲染框架(ssr)
做什么？
ssr带来的好处无非就两点SEO和首屏渲染

提升开发体验，提高用户体验

将来还会怎么更好用？
服务端支持引入dll
支持其他框架服务端渲染，如vue
更友好的处理异步操作方式期待react新的服务器渲染器

#!/usr/bin/env node
var option = require('minimist')(process.argv.slice(2))
var spawn = require('child_process').spawn

var { GlobalConfig, GlobalConfigParse, GlobalHelp, GlobalUsage} = require('./global-config')
var BuildTool = require('./options/build')
var BundleTool = require('./options/bundle')
var PatchTool = require('./options/patch')

function printHelp(){
console.log(GlobalUsage)
console.log(GlobalHelp)
}

function command(){
var commands = option._
if(!commands || commands.length < 1){
console.log(GlobalUsage)
console.log(GlobalHelp)
return
}
var command = commands[0]
try{
GlobalConfigParse(option)
}catch(err) {
printHelp()
throw err
}
try{
if(command === 'build' || command === 'build-release') {
BuildTool.execute(option, command === 'build-release')
} else if(command === 'bundle'){
BundleTool.execute()
} else if(command === 'patch') {
PatchTool.execute(option)
}else{
printHelp()
}
}catch(err) {
console.log(err)
//throw err
console.log('==============================build failed================================')
process.exitCode = 1
process.exit(1)
process.kill(process.pid, 'SIGHUP')
}
}
