## React
```javascript
class Test extends Component {
    state = {
        text: 1
    }

    toggleText = () => {
        const { text } = this.state
        const newText = text === 1 ? 2: 1 
        this.setState({ text: newText })
    }

    renderButton() {
        const { text } = this.state

        return (
            <div onClick={this.toggleText}>
                <p>{text}</p>
            </div>
        )
    }

    renderContent() {
        if(this.content) return this.content

        this.content = (
            <div>
                {this.renderButton()}
            </div>
        )
        return this.content
    }
    
    render() {
        <div>
            {this.renderContent()}
        </div>
    }
}

// 这个组件第一次render后会保存起来到this.content，之后子组件点击setState后你们说renderButton里的text值会变化
```
